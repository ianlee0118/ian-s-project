let r;
let g;
let b;

let goX = 3;
let y = 0;
let goY = 7;

  createCanvas(500, 500);
}
 
function draw() {
  
  background(0, 10);
  
  r=random(255);
  g=random(255);
  b=random(255);
  
  ellipseX = random(500);
  ellipseY = random(500);
  ellipseW = random(20,50);
  ellipseH = random(20,50);

  rect(ellipseX,ellipseY,ellipseW);
  
  fill("blue");
  stroke(255);
  ellipse(x,y, 50,50);
  
  x = x + goX;
  y = y + goY;
  
  if(x> 500-25){

  }

  }
  if(y>500-25){
    goY = -7;
  }

    goY = 9;
  }
  }
