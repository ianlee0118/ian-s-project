var sketch_1 = function(tril){
     var xspeed = 3;
    var yspeed = 5;
    var d = 25;
    var x = d;
    var y = d;
 
    tril.setup = function() {
        var can = tril.createCanvas(500, 300);
        can.position(0, 0);
    }
    
    tril.draw = function() {
        tril.background(220);
      
        tril.ellipse(x,y,d,d);
      
        x = x + xspeed;
        y = y + yspeed;
      
        if(x > tril.width-d/2 || x < d/2){
            xspeed = xspeed *-1;
      }
      if(y > tril.height-d/2 || y < d/2){
        yspeed = yspeed *-1;
      }
      
    }
}
